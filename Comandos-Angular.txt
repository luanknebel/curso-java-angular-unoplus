#instalar versão do angular 14
npm install -g @angular/cli@14.0.0

#Instalar bibliotecas
npm install angular-notifier --save
npm install @angular/material@13 --save
npm install material-icons --save
npm install ng2-currency-mask --save

#Criar serviço rest
ng generate service core/rest-client/rest-client
ng generate interceptor core/rest-client/htp-request

#Criar serviço de notificação
ng generate service core/service/notifier-handler

#Criar componentes
ng generate component login
ng generate component cadastro/cadastro-produto
